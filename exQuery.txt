SELECT g.g_seq, g_tag, g_readcnt, SUM(f.f_downcnt) AS f_downcnt
FROM gallery g
JOIN files f 
WHERE g.g_seq = f.g_seq AND g_tag LIKE '%동물%'
GROUP BY g.g_seq

//동물 태그가 포함된 게시글의 종합 다운로드 수
SELECT SUM(f.f_downcnt) AS f_downcnt
FROM gallery g
JOIN files f 
WHERE g.g_seq = f.g_seq AND g_tag LIKE '%동물%'


//각 태그별 조회수
SELECT t_name, SUM(t_readcnt) AS t_readcnt
FROM tag
GROUP BY t_name
ORDER BY t_readcnt DESC;

//특정 태그의 조회수
SELECT t_name, SUM(t_readcnt) AS t_readcnt
FROM tag
GROUP BY t_name
WHERE t_name LIKE '%동물%'
